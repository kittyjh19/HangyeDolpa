create sequence USER_NO_seq nocache;
create sequence CLIMB_NO_SEQ nocache;
create sequence BOARD_NO_seq nocache;
create sequence CNO_seq nocache;

CREATE TABLE USER_TABLE (
    USER_NO INT(10) PRIMARY KEY,
    AUTH_ID VARCHAR(200) NOT NULL UNIQUE,
    USER_NAME VARCHAR(200) NOT NULL,
    USER_IMG VARCHAR(200) NOT NULL
);

CREATE TABLE EXERCISE_TABLE (
    USER_NO INT(15),
    CLIMB_NO INT(10)  PRIMARY KEY, 				
    CLIMB_DATE VARCHAR(100) NOT NULL,                    
    CLIMB_PLACE VARCHAR(200) NOT NULL,           
    CLIMB_STAGE INT(10) NOT NULL,                  
    CLIMB_COUNT INT(10) NOT NULL,     
    CLIMB_TIME INT(10) NOT NULL,
	CLIMB_KCAL INT(10) NOT NULL                   
);

CREATE TABLE BOARD_TABLE (
    BOARD_NO INT(10) PRIMARY KEY,
    BOARD_TITLE VARCHAR(200) NOT NULL,
    BOARD_CONTENT VARCHAR(2000) NOT NULL,
    BOARD_REGDATE DATE DEFAULT SYSDATE(),
    BOARD_UPDATEDATE DATE DEFAULT SYSDATE(),
    USER_ID INT(200) NOT NULL,
    BOARD_CNT INT(10) NOT NULL DEFAULT 0
);

CREATE TABLE COMMENTS_TABLE (
    CNO INT(10) PRIMARY KEY,
    BOARD_NO INT(10) NOT NULL,
    USER_ID INT(10) NOT NULL,
    CONTENT VARCHAR(500) NOT NULL,
    REGDATE DATE DEFAULT SYSDATE()
);

CREATE TABLE BOARD_ATTACH (
	FILENAME VARCHAR(200) NOT NULL,
	UUID VARCHAR(200) PRIMARY KEY,
	UPLOADPATH VARCHAR(200) NOT NULL,
	IMAGE VARCHAR(1) ,
	VIDEO VARCHAR(1) ,
	BOARD_NO INT(10)
);
